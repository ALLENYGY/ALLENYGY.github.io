GETminmun(A):
  SORT(A)
  if(A[0]-A[A.length-1]<0) return A[0]-A[A.length-1];
  else{
    min=INT_MAX;
    for i from 1 to A.length-1:
      if(A[i]-A[i-1]<min) min=A[i]-A[i-1];
  }
  return min;
